420  uint32_t toDaysSinceEpochIgnoringTimeZone() const {
421  // From wiki article:
422  //
423  // JDN = (1461 x (Y + 4800 + (M - 14)/12))/4
424  // + (367 x (M - 2 - 12 x ((M - 14)/12)))/12
425  // - (3 x ((Y + 4900 + (M - 14)/12)/100))/4
426  // + D - 32075
427  // JDN2000 = JDN - 2451545
428  //
429  // It looks like the formula needs to be done using signed integers
430  // because it depends on the modulo operation (%) to truncate towards 0
431  // for negative numbers.
432 
433  int8_t mm = (mMonth - 14)/12;
434  int16_t yy = mYear + kEpochYear;
435  int32_t jdn = ((int32_t) 1461 * (yy + 4800 + mm))/4
436  + (367 * (mMonth - 2 - 12 * mm))/12
437  - (3 * ((yy + 4900 + mm)/100))/4
438  + mDay - 32075;
439  return jdn - kDaysSinceJulianEpoch;
440  }

449  void fillUsingDaysSinceEpoch(uint32_t daysSinceEpoch) {
450  uint32_t J = daysSinceEpoch + kDaysSinceJulianEpoch;
451  uint32_t f = J + 1401 + (((4 * J + 274277 ) / 146097) * 3) / 4 - 38;
452  uint32_t e = 4 * f + 3;
453  uint32_t g = e % 1461 / 4;
454  uint32_t h = 5 * g + 2;
455  mDay = (h % 153) / 5 + 1;
456  mMonth = (h / 153 + 2) % 12 + 1;
457  mYear = (e / 1461) - 4716 + (12 + 2 - mMonth) / 12 - kEpochYear;
458 
459  // 2000-01-01 is Saturday (7)
460  mDayOfWeek = (daysSinceEpoch + 6) % 7 + 1;
461  }
