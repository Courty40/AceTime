122  virtual uint32_t readResponse() const override {
123  // read packet into the buffer
124  mUdp.read(mPacketBuffer, kNtpPacketSize);
125 
126  // convert four bytes starting at location 40 to a long integer
127  uint32_t secsSince1900 = (uint32_t) mPacketBuffer[40] << 24;
128  secsSince1900 |= (uint32_t) mPacketBuffer[41] << 16;
129  secsSince1900 |= (uint32_t) mPacketBuffer[42] << 8;
130  secsSince1900 |= (uint32_t) mPacketBuffer[43];
131  return (secsSince1900 == 0) ? 0 : secsSince1900 - kSecondsSinceNtpEpoch;
132  }

145  void sendNtpPacket(const IPAddress& address) const {
146 #if ACE_TIME_NTP_TIME_PROVIDER_DEBUG == 1
147  uint16_t startTime = millis();
148 #endif
149  // set all bytes in the buffer to 0
150  memset(mPacketBuffer, 0, kNtpPacketSize);
151  // Initialize values needed to form NTP request
152  // (see URL above for details on the packets)
153  mPacketBuffer[0] = 0b11100011; // LI, Version, Mode
154  mPacketBuffer[1] = 0; // Stratum, or type of clock
155  mPacketBuffer[2] = 6; // Polling Interval
156  mPacketBuffer[3] = 0xEC; // Peer Clock Precision
157  // 8 bytes of zero for Root Delay & Root Dispersion
158  mPacketBuffer[12] = 49;
159  mPacketBuffer[13] = 0x4E;
160  mPacketBuffer[14] = 49;
161  mPacketBuffer[15] = 52;
162  // all NTP fields have been given values, now
163  // you can send a packet requesting a timestamp:
164  mUdp.beginPacket(address, 123); //NTP requests are to port 123
165  mUdp.write(mPacketBuffer, kNtpPacketSize);
166  mUdp.endPacket();
167 #if ACE_TIME_NTP_TIME_PROVIDER_DEBUG == 1
168  common::logger("NtpTimeProvider::sendNtpPacket(): %u ms",
169  (uint16_t) (millis() - startTime));
170 #endif
171  }
