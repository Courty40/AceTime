312  // Make sure that mSecondsSinceEpoch does not fall too far behind.
313  if (timeSinceLastSync >= 5000) {
314  mSystemTimeKeeper.getNow();
315  }
316 
317  // Synchronize if a TimeProvider is available, and mSyncPeriodSeconds has

320  if (timeSinceLastSync >= mSyncPeriodSeconds * (uint32_t) 1000) {
321  // blocking call

323  if (nowSeconds == 0) return;
