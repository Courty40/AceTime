73 inline int acetime_strcmp_PP(const char* a, const char* b) {
74  if (a == b) { return 0; }
75  if (a == nullptr) { return -1; }
76  if (b == nullptr) { return 1; }
77 
78  while (true) {
79  uint8_t ca = pgm_read_byte(a);
80  uint8_t cb = pgm_read_byte(b);
81  if (ca != cb) return (int) ca - (int) cb;
82  if (ca == '\0') return 0;
83  a++;
84  b++;
85  }
86 }
